        function sendMessage() {
            const input = document.getElementById('chatInput');
            let message = input.value.trim().toLowerCase();
            if(message === "") return;

            const chatHistory = document.getElementById('chatHistory');

            const userMsg = document.createElement('p');
            userMsg.textContent = input.value.trim();
            chatHistory.appendChild(userMsg);

            message = message.replace(/\s+/g, ' ');

            let response = respuestasJoel[message] || "No puedo responder a eso.";

            const joelMsg = document.createElement('p');
            joelMsg.textContent = response;
            joelMsg.classList.add('joel');
            chatHistory.appendChild(joelMsg);

            input.value = "";

            chatHistory.scrollTop = chatHistory.scrollHeight;
        }

        document.getElementById('chatInput').addEventListener('keypress', function(e) {
            if(e.key === 'Enter') {
                sendMessage();
            }
        });
